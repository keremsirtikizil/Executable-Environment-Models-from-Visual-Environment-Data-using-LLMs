Round 14  —  frames [39, 40, 41]
==================================================

STATE VARIABLES:
  Mechanism.state: {LOCKED_VERTICAL, UNBOLTED_FOR_FOLDING, FOLDED, LOCKED_HORIZONTAL, ROTATING}
  UserAction: {UNLOCKS_BOLTS, LOCKS_BOLTS, FOLDS_BRACE, UNFOLDS_BRACE, UNLOCKS_KEY, LOCKS_KEY, MOVES_PANELS, NONE}
  PanelLeft.isMobile: {TRUE, FALSE}
  PanelRight.isMobile: {TRUE, FALSE}
  PanelGap.width: float

RULES:
  // Disassembly Path
  IF Mechanism.state == LOCKED_VERTICAL AND UserAction == UNLOCKS_BOLTS
    THEN SET Mechanism.state = UNBOLTED_FOR_FOLDING

  IF Mechanism.state == UNBOLTED_FOR_FOLDING AND UserAction == FOLDS_BRACE
    THEN SET Mechanism.state = FOLDED

  IF Mechanism.state == FOLDED AND UserAction == MOVES_PANELS
    THEN CHANGE PanelGap.width

  IF Mechanism.state == FOLDED AND UserAction == UNFOLDS_BRACE
    THEN SET Mechanism.state = UNBOLTED_FOR_FOLDING

  IF Mechanism.state == UNBOLTED_FOR_FOLDING AND UserAction == LOCKS_BOLTS
    THEN SET Mechanism.state = LOCKED_VERTICAL

  // Door Path
  IF Mechanism.state == LOCKED_VERTICAL AND UserAction == UNLOCKS_KEY
    THEN SET Mechanism.state = ROTATING

  IF Mechanism.state == ROTATING AND UserAction == LOCKS_KEY AND Brace.angle is near 90
    THEN SET Mechanism.state = LOCKED_HORIZONTAL

  IF Mechanism.state == LOCKED_HORIZONTAL AND UserAction == UNLOCKS_KEY
    THEN SET Mechanism.state = ROTATING

  // Panel Mobility Logic
  IF Mechanism.state == LOCKED_VERTICAL OR Mechanism.state == UNBOLTED_FOR_FOLDING
    THEN SET PanelLeft.isMobile = FALSE
    THEN SET PanelRight.isMobile = FALSE
  ELSE IF Mechanism.state == FOLDED
    THEN SET PanelLeft.isMobile = TRUE
    THEN SET PanelRight.isMobile = TRUE
  ELSE IF Mechanism.state == LOCKED_HORIZONTAL OR Mechanism.state == ROTATING
    THEN SET PanelLeft.isMobile = FALSE
    THEN SET PanelRight.isMobile = TRUE

    THEN SET System.state = ROTATABLE_DOOR
  IF System.state == ROTATABLE_DOOR AND Brace.orientation == VERTICAL AND UserAction == TOGGLE_KEY_LOCK
    THEN SET System.state = LOCKED_WALL
  IF System.state == ROTATABLE_DOOR AND Brace.orientation == HORIZONTAL AND UserAction == TOGGLE_KEY_LOCK
    THEN SET System.state = LOCKED_DOOR
  IF System.state == LOCKED_DOOR AND UserAction == TOGGLE_KEY_LOCK
    THEN SET System.state = ROTATABLE_DOOR

  IF System.state == LOCKED_WALL AND UserAction == TOGGLE_BOLTS
    THEN SET System.state = UNBOLTED_WALL
  IF System.state == UNBOLTED_WALL AND UserAction == TOGGLE_FOLD
    THEN SET System.state = FOLDED_WALL
  IF System.state == FOLDED_WALL AND UserAction == TOGGLE_FOLD
    THEN SET System.state = UNBOLTED_WALL
  IF System.state == UNBOLTED_WALL AND UserAction == TOGGLE_BOLTS
    THEN SET System.state = LOCKED_WALL

  IF System.state == FOLDED_WALL
    THEN SET PanelLeft.isMobile = TRUE
    THEN SET PanelRight.isMobile = TRUE
    THEN SET PanelRight.isSwingable = FALSE
  ELSE IF System.state == ROTATABLE_DOOR OR System.state == LOCKED_DOOR
    THEN SET PanelLeft.isMobile = FALSE
    THEN SET PanelRight.isMobile = FALSE
    THEN SET PanelRight.isSwingable = TRUE
  ELSE // State is LOCKED_WALL or UNBOLTED_WALL
    THEN SET PanelLeft.isMobile = FALSE
    THEN SET PanelRight.isMobile = FALSE
    THEN SET PanelRight.isSwingable = FALSE
