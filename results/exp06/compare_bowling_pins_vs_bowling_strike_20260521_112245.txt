STATE VARIABLES:
  simulation_state: (active, finished)
  game_state: (ready, ball_rolling, pins_interacting, end_of_roll)
  reset_cycle: (idle, preparing_spare, resetting_frame)
  roll_number: (1, 2)
  game_is_over: BOOLEAN
  initial_hit_type: (none, pocket, brooklyn, nose_hit)
  ball: Ball object with properties (position, velocity)
  pins: LIST of Pin objects, each with properties (id, position, state: {standing, falling, fallen, lifted})
  sweep_arm: SweepArm object with properties (state: {retracted, lowering, sweeping, retracting})
  pinsetter_deck: PinsetterDeck object with properties (state: {raised, lowering, on_deck, lifting, spotting, raising})
  ui_text: STRING

RULES:
  IF simulation_state IS finished:
    THEN RENDER black_screen
    // No other rules are processed

  ELSE IF simulation_state IS active:
    IF game_state IS ready AND a ball is detected entering the lane:
      THEN SET game_state to ball_rolling
      SET initial_hit_type to none
      CLEAR ui_text

    IF game_state IS ball_rolling AND ball collides_with a pin for the first time:
      THEN SET game_state to pins_interacting
      IF first pin hit is pin_1 AND ball.impact_point is on the left side of pin_1:
        THEN SET initial_hit_type to brooklyn
      ELSE IF first pin hit is pin_1 AND ball.impact_point is on the right side of pin_1:
        THEN SET initial_hit_type to pocket
      ELSE IF first pin hit is pin_1 AND ball.impact_point is near the center of pin_1:
        THEN SET initial_hit_type to nose_hit
      TRANSFER momentum from ball to the hit pin(s)
      SET hit_pin(s).state to falling

    IF game_state IS pins_interacting:
      UPDATE ball.position and ball.velocity
      FOR EACH pin_A in pins:
        IF pin_A.state IS falling:
          UPDATE pin_A.position and pin_A.velocity
          FOR EACH pin_B in pins where pin_A IS NOT pin_B:
            IF pin_A collides_with pin_B AND pin_B.state IS standing:
              THEN TRANSFER momentum from pin_A to pin_B
              SET pin_B.state to falling
      IF all pins and ball have stopped moving:
        THEN SET game_state to end_of_roll

    IF game_state IS end_of_roll AND reset_cycle IS idle:
      LET standing_pins = GET_PINS_BY_STATE(standing)
      IF roll_number IS 1:
        IF COUNT(standing_pins) IS 0: // Strike
          IF initial_hit_type IS brooklyn: THEN SET ui_text to "Brooklyn Strike"
          ELSE: THEN SET ui_text to "Strike"
          SET reset_cycle to resetting_frame
        ELSE: // Open first roll
          SET reset_cycle to preparing_spare
      ELSE IF roll_number IS 2:
        SET reset_cycle to resetting_frame
      IF game_completion_condition is met: // e.g., 10th frame is over
        SET game_is_over to TRUE
      DISPLAY ui_text

    IF reset_cycle IS preparing_spare:
      IF pinsetter_deck.state IS raised: THEN SET pinsetter_deck.state to lowering
      IF pinsetter_deck.state IS on_deck:
        FOR EACH pin in pins WHERE pin.state IS standing: SET pin.state to lifted
        SET pinsetter_deck.state to lifting
      IF pinsetter_deck.state IS lifting AND deck is clear of sweep path: SET sweep_arm.state to lowering
      IF sweep_arm.state IS lowering AND arm reaches lane: SET sweep_arm.state to sweeping
      IF sweep_arm.state IS sweeping:
        MOVE sweep_arm forward; REMOVE all pins WHERE state IS fallen
        IF sweep_arm reaches end of deck: SET sweep_arm.state to retracting
      IF sweep_arm.state IS retracted AND pinsetter_deck.state IS lifting: SET pinsetter_deck.state to spotting
      IF pinsetter_deck.state IS spotting AND deck is on_deck:
        FOR EACH pin in pins WHERE pin.state IS lifted: SET pin.state to standing
        SET pinsetter_deck.state to raising
      IF pinsetter_deck.state IS raised:
        SET reset_cycle to idle; SET game_state to ready; SET roll_number to 2

    IF reset_cycle IS resetting_frame:
      IF sweep_arm.state IS retracted: SET sweep_arm.state to lowering
      IF sweep_arm.state IS lowering AND arm reaches lane: SET sweep_arm.state to sweeping
      IF sweep_arm.state IS sweeping:
        MOVE sweep_arm forward; REMOVE all pins from lane
        IF sweep_arm reaches end of deck: SET sweep_arm.state to retracting
      IF sweep_arm.state IS retracted AND pinsetter_deck.state IS raised: SET pinsetter_deck.state to lowering
      IF pinsetter_deck.state IS on_deck:
        CREATE new set of 10 standing pins
        SET pinsetter_deck.state to raising
      IF pinsetter_deck.state IS raised:
        IF game_is_over IS TRUE:
          THEN SET simulation_state to finished
        ELSE:
          SET reset_cycle to idle; SET game_state to ready; SET roll_number to 1

PSEUDOCODE B — bowling_strike
==================================================

STATE VARIABLES:
  game_state: {AWAITING_ROLL, BALL_IN_PLAY, POST_ROLL, SWEEPING, RESETTING}
  roll_number: {1, 2}
  first_roll_score: number
  score_display.text: string
  ball.position: (x, z)
  ball.velocity: (vx, vz)
  ball.visible: boolean
  pins: array of pin objects
  pin.id: number
  pin.position: (x, z)
  pin.initial_position: (x, z)
  pin.state: {standing, falling, fallen, cleared}
  pin.velocity: (vx, vz)
  pin.angular_velocity: number
  sweep_bar.is_active: boolean
  pins_are_settled: boolean
  NUM_PINS: constant
  COLLISION_DISTANCE: constant
  PIN_DECK_END_Z: constant

RULES:
  IF game_state IS AWAITING_ROLL:
    // On user input to roll:
    game_state = BALL_IN_PLAY
    ball.visible = true
    ball.position = ball.initial_position
    ball.velocity = ball.initial_velocity

  IF game_state IS BALL_IN_PLAY:
    ball.position = ball.position + ball.velocity
    IF ball.position.z > PIN_DECK_END_Z:
      ball.visible = false
      game_state = POST_ROLL
      pins_are_settled = false
    ELSE:
      FOR EACH pin IN pins:
        IF pin.state IS standing AND distance(ball.position, pin.position) < COLLISION_DISTANCE:
          pin.state = falling
          pin.velocity = calculate_imparted_velocity(ball, pin)
          pin.angular_velocity = calculate_imparted_angular_velocity(ball, pin)

  IF game_state IS BALL_IN_PLAY OR game_state IS POST_ROLL:
    is_any_pin_moving = false
    FOR EACH pin1 IN pins:
      IF pin1.state IS falling:
        is_any_pin_moving = true
        pin1.position = pin1.position + pin1.velocity
        // Check for pin-pin collisions
        FOR EACH pin2 IN pins:
          IF pin1.id != pin2.id AND pin2.state IS standing AND distance(pin1.position, pin2.position) < COLLISION_DISTANCE:
            pin2.state = falling
            pin2.velocity = calculate_imparted_velocity(pin1, pin2)
            pin2.angular_velocity = calculate_imparted_angular_velocity(pin1, pin2)
        // Check for settling
        IF pin1.velocity is near zero:
          pin1.state = fallen
    IF game_state IS POST_ROLL AND is_any_pin_moving IS false:
      pins_are_settled = true

  IF game_state IS POST_ROLL AND pins_are_settled IS true:
    fallen_this_roll = count(pins where state IS fallen OR state IS falling)
    IF roll_number IS 1:
      first_roll_score = fallen_this_roll
      score_display.text = first_roll_score
      IF first_roll_score == NUM_PINS: // Strike
        game_state = RESETTING
      ELSE:
        game_state = SWEEPING
    ELSE: // roll_number IS 2
      second_roll_score = fallen_this_roll
      total_frame_score = first_roll_score + second_roll_score
      score_display.text = total_frame_score
      game_state = RESETTING

  IF game_state IS SWEEPING:
    IF sweep_bar.is_active IS false:
      sweep_bar.is_active = true
    // Sweeping action occurs
    FOR EACH pin IN pins:
      IF pin.state IS fallen:
        pin.state = cleared
    // Assume sweep completes
    sweep_bar.is_active = false
    roll_number = 2
    game_state = AWAITING_ROLL

  IF game_state IS RESETTING:
    // After a delay:
    FOR EACH pin IN pins:
      pin.state = standing
      pin.position = pin.initial_position
    roll_number = 1
    first_roll_score = 0
    score_display.text = ""
    game_state = AWAITING_ROLL
