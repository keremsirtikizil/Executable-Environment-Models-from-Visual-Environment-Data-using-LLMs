    THEN `ball1.y` = `INITIAL_Y1`

    THEN `ball2.y` = `ball2.y` + `ball2.velocityY`
    THEN `ball1.temperature` = `max`(`AMBIENT_TEMP`, `ball1.temperature` - `COOLING_RATE`)
    THEN `ball2.temperature` = `max`(`AMBIENT_TEMP`, `ball2.temperature` - `COOLING_RATE`)
    THEN FOR EACH `fragment` IN `fragments`:
      THEN `fragment.x` = `fragment.x` + `fragment.velocityX`
      THEN `fragment.y` = `fragment.y` + `fragment.velocityY`
      THEN `fragment.lifespan` = `fragment.lifespan` - 1
